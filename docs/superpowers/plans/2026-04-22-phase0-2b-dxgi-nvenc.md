```rust
//! DXGI Output (monitor) enumeration.

use windows::core::Interface;
use windows::Win32::Foundation::RECT;
use windows::Win32::Graphics::Dxgi::{
    IDXGIAdapter1, IDXGIFactory1, IDXGIOutput, DXGI_OUTPUT_DESC,
    CreateDXGIFactory1,
};

use crate::adapter::AdapterInfo;
use crate::error::{MediaError, Result};

#[derive(Debug, Clone)]
pub struct OutputInfo {
    pub index: u32,
    pub device_name: String,
    pub desktop_rect: RECT,
    pub rotation: u32, // DXGI_MODE_ROTATION enum value
    pub is_attached: bool,
}

pub fn enumerate_outputs_for_adapter(adapter: &AdapterInfo) -> Result<Vec<OutputInfo>> {
    unsafe {
        let factory: IDXGIFactory1 =
            CreateDXGIFactory1().map_err(|e| MediaError::dxgi("CreateDXGIFactory1", e))?;
        let adapter_com: IDXGIAdapter1 = factory
            .EnumAdapters1(adapter.index)
            .map_err(|e| MediaError::dxgi("EnumAdapters1 (output enum)", e))?;

        let mut out = Vec::new();
        let mut i = 0u32;
        loop {
            let output: IDXGIOutput = match adapter_com.EnumOutputs(i) {
                Ok(o) => o,
                Err(e) if e.code() == windows::Win32::Graphics::Dxgi::DXGI_ERROR_NOT_FOUND => break,
                Err(e) => return Err(MediaError::dxgi("EnumOutputs", e)),
            };
            let desc: DXGI_OUTPUT_DESC = output
                .GetDesc()
                .map_err(|e| MediaError::dxgi("IDXGIOutput::GetDesc", e))?;
            let name = String::from_utf16_lossy(
                &desc.DeviceName[..desc
                    .DeviceName
                    .iter()
                    .position(|&c| c == 0)
                    .unwrap_or(desc.DeviceName.len())],
            );
            out.push(OutputInfo {
                index: i,
                device_name: name,
                desktop_rect: desc.DesktopCoordinates,
                rotation: desc.Rotation.0 as u32,
                is_attached: desc.AttachedToDesktop.as_bool(),
            });
            i += 1;
        }
        Ok(out)
    }
}
```

Note: adjust `AttachedToDesktop.as_bool()` if the 0.58 windows crate uses a different BOOL→bool API (might be `.into()` or field access).

```rust
//! DXGI Desktop Duplication wrapper.

use std::time::Duration;

use windows::core::Interface;
use windows::Win32::Graphics::Direct3D11::{ID3D11Resource, ID3D11Texture2D};
use windows::Win32::Graphics::Dxgi::{
    IDXGIDevice, IDXGIOutput1, IDXGIOutputDuplication, IDXGIResource, DXGI_ERROR_ACCESS_LOST,
    DXGI_ERROR_WAIT_TIMEOUT, DXGI_OUTDUPL_FRAME_INFO,
};

use crate::d3d11::{D3d11Device, D3d11Texture, TextureFormat};
use crate::dxgi::output::OutputInfo;
use crate::error::{MediaError, Result};

/// Outcome of a single `acquire_next_frame` call.
pub enum AcquiredFrame {
    /// A new frame arrived. Caller must call `release()` after processing.
    Frame { texture: D3d11Texture, frame_info: DXGI_OUTDUPL_FRAME_INFO },
    /// No new frame within the timeout window. The desktop didn't change.
    Timeout,
}

pub struct DesktopDuplication {
    dup: IDXGIOutputDuplication,
    width: u32,
    height: u32,
    frame_held: bool,
}

impl DesktopDuplication {
    pub fn new(dev: &D3d11Device, output: &OutputInfo) -> Result<Self> {
        unsafe {
            // Get IDXGIDevice from the D3D11 device.
            let dxgi_device: IDXGIDevice = dev
                .device()
                .cast()
                .map_err(|e| MediaError::dxgi("D3D11 device -> IDXGIDevice cast", e))?;

            // Re-enumerate to get IDXGIOutput1 at the same index.
            use windows::Win32::Graphics::Dxgi::{
                CreateDXGIFactory1, IDXGIAdapter1, IDXGIFactory1, IDXGIOutput,
            };
            let factory: IDXGIFactory1 =
                CreateDXGIFactory1().map_err(|e| MediaError::dxgi("CreateDXGIFactory1", e))?;
            let adapter: IDXGIAdapter1 = factory
                .EnumAdapters1(dev.adapter().index)
                .map_err(|e| MediaError::dxgi("EnumAdapters1", e))?;
            let output_base: IDXGIOutput = adapter
                .EnumOutputs(output.index)
                .map_err(|e| MediaError::dxgi("EnumOutputs", e))?;
            let output1: IDXGIOutput1 = output_base
                .cast()
                .map_err(|e| MediaError::dxgi("IDXGIOutput -> IDXGIOutput1 cast", e))?;

            let dup: IDXGIOutputDuplication = output1
                .DuplicateOutput(&dxgi_device)
                .map_err(|e| MediaError::dxgi("DuplicateOutput", e))?;

            // Pull dimensions from the output desc.
            let width = (output.desktop_rect.right - output.desktop_rect.left) as u32;
            let height = (output.desktop_rect.bottom - output.desktop_rect.top) as u32;

            Ok(Self { dup, width, height, frame_held: false })
        }
    }

    pub fn width(&self) -> u32 { self.width }
    pub fn height(&self) -> u32 { self.height }

    /// Acquire the next duplicated frame. `timeout` is clamped to u32 milliseconds.
    pub fn acquire_next_frame(&mut self, timeout: Duration) -> Result<AcquiredFrame> {
        if self.frame_held {
            // Release the previously held frame before acquiring a new one.
            unsafe {
                self.dup.ReleaseFrame()
                    .map_err(|e| MediaError::dxgi("ReleaseFrame (before acquire)", e))?;
            }
            self.frame_held = false;
        }

        let ms = timeout.as_millis().min(u32::MAX as u128) as u32;
        let mut info = DXGI_OUTDUPL_FRAME_INFO::default();
        let mut resource: Option<IDXGIResource> = None;

        let ret = unsafe { self.dup.AcquireNextFrame(ms, &mut info, &mut resource) };
        match ret {
            Ok(()) => {}
            Err(e) if e.code() == DXGI_ERROR_WAIT_TIMEOUT => return Ok(AcquiredFrame::Timeout),
            Err(e) if e.code() == DXGI_ERROR_ACCESS_LOST => {
                return Err(MediaError::Dxgi {
                    context: "AcquireNextFrame (access lost)",
                    hresult: DXGI_ERROR_ACCESS_LOST.0 as u32,
                });
            }
            Err(e) => return Err(MediaError::dxgi("AcquireNextFrame", e)),
        }
        self.frame_held = true;

        let resource = resource.ok_or(MediaError::Dxgi {
            context: "AcquireNextFrame returned null IDXGIResource",
            hresult: 0,
        })?;

        let texture_com: ID3D11Texture2D = resource
            .cast()
            .map_err(|e| MediaError::d3d11("IDXGIResource -> ID3D11Texture2D cast", e))?;

        let tex = D3d11Texture::from_raw(texture_com, self.width, self.height, TextureFormat::Bgra8);
        Ok(AcquiredFrame::Frame { texture: tex, frame_info: info })
    }

    pub fn release_frame(&mut self) -> Result<()> {
        if self.frame_held {
            unsafe {
                self.dup.ReleaseFrame()
                    .map_err(|e| MediaError::dxgi("ReleaseFrame", e))?;
            }
            self.frame_held = false;
        }
        Ok(())
    }
}

impl Drop for DesktopDuplication {
    fn drop(&mut self) {
        let _ = self.release_frame();
    }
}

// Note: ID3D11Resource is imported but not directly used — silence warning.
const _: Option<ID3D11Resource> = None;
```

### Task 11: `nvenc::loader` — runtime DLL load

Create `crates/media-win/src/nvenc/loader.rs`:
```rust
//! Runtime loader for `nvEncodeAPI64.dll`. The NVIDIA docs mandate using
//! LoadLibrary + GetProcAddress rather than static linking.

use std::ffi::c_void;
use std::sync::OnceLock;

use libloading::Library;

use crate::error::{MediaError, Result};

pub struct NvEncLibrary {
    _lib: Library,
    create_instance: unsafe extern "C" fn(*mut c_void) -> i32,
}

static LIB: OnceLock<Result<NvEncLibrary>> = OnceLock::new();

impl NvEncLibrary {
    pub fn load() -> Result<&'static NvEncLibrary> {
        LIB.get_or_init(|| unsafe {
            let lib = Library::new("nvEncodeAPI64.dll")
                .map_err(|e| MediaError::Other(format!("LoadLibrary nvEncodeAPI64.dll: {e}")))?;
            let create_instance: libloading::Symbol<unsafe extern "C" fn(*mut c_void) -> i32> =
                lib.get(b"NvEncodeAPICreateInstance\0")
                    .map_err(|e| MediaError::Other(format!("GetProcAddress: {e}")))?;
            let fp = *create_instance;
            Ok(NvEncLibrary {
                _lib: lib,
                create_instance: fp,
            })
        }).as_ref().map_err(|e| e.clone_for_ref())
    }

    pub fn create_instance(&self, fn_table: *mut c_void) -> i32 {
        unsafe { (self.create_instance)(fn_table) }
    }
}

// Helper — MediaError is not Clone by default; we provide a shallow clone for
// static-error propagation. (Alternative: use Arc<MediaError>.)
impl MediaError {
    pub(crate) fn clone_for_ref(&self) -> MediaError {
        match self {
            Self::Other(s) => Self::Other(s.clone()),
            other => Self::Other(format!("{other}")),
        }
    }
}
```

### Task 12-14: Config, Encoder, Smoke Test

     - Fill `NV_ENC_PIC_PARAMS` (including `encodePicFlags |= NV_ENC_PIC_FLAG_FORCEIDR` when IDR requested)
     - `EncodePicture` → async until complete
     - `LockBitstream` → copy NAL bytes → `UnlockBitstream`
     - `UnmapInputResource`
  5. `DestroyEncoder` on drop
- **Task 14 (nvenc_smoke.rs)**:
  - Use `synthetic::make_counter_texture` to produce an input frame.
  - Encode it with `NvencEncoder`.
  - Verify output has H.265 NAL start code `00 00 00 01` and VPS/SPS/PPS NAL unit types (32/33/34) in the first IDR.

## Known Risks for Plan 2b

1. **NVIDIA SDK version skew**: struct fields have been added/removed across SDK 11→12→13. The `NV_ENC_INITIALIZE_PARAMS` `version` field must match `NV_ENC_INITIALIZE_PARAMS_VER` from the installed SDK.
2. **D3D11 device type mismatch**: NVENC requires the D3D11 device to be from a supported adapter. On GTX 1080 HEVC encode is supported; on older cards might fail.
3. **ReleaseFrame pairing**: If `acquire_next_frame` is called without releasing the previous, DXGI returns `DXGI_ERROR_INVALID_CALL`. The Drop impl handles this.
4. **UAC elevation**: DXGI Desktop Duplication cannot capture elevated windows (UAC prompt, Task Manager with elevation). Normal behavior.
5. **HDR output**: Desktop Duplication on HDR-enabled displays returns FP16 format, not BGRA8. Plan 2b assumes SDR; HDR support is deferred.
